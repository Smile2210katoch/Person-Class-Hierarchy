import React from "react";
import "./App.css";

// Base Class: Person
class Person {
  constructor(name, department) {
    this.name = name;
    this.department = department;
  }

  displayInfo() {
    return `Name: ${this.name}, Department: ${this.department}`;
  }
}

// Subclass: Student
class Student extends Person {
  constructor(name, age, course) {
    super(name, "Student Dept"); // or you can pass a specific department if needed
    this.age = age;
    this.course = course;
  }

  displayInfo() {
    return `Student Name: ${this.name}, Age: ${this.age}, Course: ${this.course}`;
  }
}

// Subclass: Teacher
class Teacher extends Person {
  constructor(name, department, subject) {
    super(name, department);
    this.subject = subject;
  }

  displayInfo() {
    return `Teacher Name: ${this.name}, Department: ${this.department}, Subject: ${this.subject}`;
  }
}

// React Component
const App = () => {
  const student1 = new Student("Smile Katoch", 20, "CSE");
  const teacher1 = new Teacher("Ms. Gurbani", "AIT", "ADBMS");

  return (
    <div className="container">
      <h1>Person Class Hierarchy</h1>
      <div className="card">{student1.displayInfo()}</div>
      <div className="card">{teacher1.displayInfo()}</div>
    </div>
  );
};

export default App;
